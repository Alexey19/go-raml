package client

import (
	"encoding/json"
	"net/http"

	"examples.com/libro/goraml"
	"examples.com/libro/types_lib"
)

type PersonService service

func (s *PersonService) PersonGet(headers, queryParams map[string]interface{}) (types_lib.Person, *http.Response, error) {
	var err error
	var respBody200 types_lib.Person

	resp, err := s.client.doReqNoBody("GET", s.client.BaseURI+"/person", headers, queryParams)
	if err != nil {
		return respBody200, nil, err
	}
	defer resp.Body.Close()

	switch resp.StatusCode {
	case 200:
		err = json.NewDecoder(resp.Body).Decode(&respBody200)
	default:
		err = goraml.NewAPIError(resp, nil)
	}

	return respBody200, resp, err
}
