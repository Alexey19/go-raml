package main

import (
	"log"
	"net/http"

	"examples.com/ramlcode/goraml"
	"examples.com/ramlcode/handlers/helloworld"
	"examples.com/ramlcode/handlers/users"
	"github.com/gorilla/mux"
	"gopkg.in/validator.v2"
)

func main() {
	// input validator
	validator.SetValidationFunc("multipleOf", goraml.MultipleOf)

	r := mux.NewRouter()

	// home page
	r.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		http.ServeFile(w, r, "index.html")
	})

	// apidocs
	r.PathPrefix("/apidocs/").Handler(http.StripPrefix("/apidocs/", http.FileServer(http.Dir("./apidocs/"))))

	HelloworldInterfaceRoutes(r, helloworld.HelloworldAPI{})

	UsersInterfaceRoutes(r, users.UsersAPI{})

	log.Println("starting server")
	http.ListenAndServe(":5000", r)
}
