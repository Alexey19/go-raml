package theclient

import (
	"encoding/json"
	"net/http"

	"examples.com/libro/goraml"
)

type Escape_typeService service

func (s *Escape_typeService) Escape_typePost(body User2_0, headers, queryParams map[string]interface{}) (User2_0, *http.Response, error) {
	var err error
	var respBody200 User2_0

	resp, err := s.client.doReqWithBody("POST", s.client.BaseURI+"/escape_type", &body, headers, queryParams)
	if err != nil {
		return respBody200, nil, err
	}
	defer resp.Body.Close()

	switch resp.StatusCode {
	case 200:
		err = json.NewDecoder(resp.Body).Decode(&respBody200)
	default:
		err = goraml.NewAPIError(resp, nil)
	}

	return respBody200, resp, err
}
