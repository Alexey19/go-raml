package users

import (
	"encoding/json"
	"examples.com/libro/types"
	"net/http"
)

// UsersAPI is API implementation of /users root endpoint
type UsersAPI struct {
}

// CreateUsers is the handler for POST /users
func (api UsersAPI) CreateUsers(w http.ResponseWriter, r *http.Request) {
	var reqBody types.User

	// decode request
	if err := json.NewDecoder(r.Body).Decode(&reqBody); err != nil {
		w.WriteHeader(400)
		return
	}

	// validate request
	if err := reqBody.Validate(); err != nil {
		w.WriteHeader(400)
		w.Write([]byte(`{"error":"` + err.Error() + `"}`))
		return
	}
}
