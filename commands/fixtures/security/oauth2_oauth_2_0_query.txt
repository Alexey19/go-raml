package main

import (
	"log"
	"net/http"
)

func oauth_2_0_queryMwr(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {

		token := r.URL.Query().Get("access_token")
		if token == "" {
			log.Println("Authorization header is empty")
			w.WriteHeader(401)
			return
		}

		next.ServeHTTP(w, r)
	})
}
